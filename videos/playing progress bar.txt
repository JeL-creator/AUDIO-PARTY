  <script>
    let currentAudio = null;
    let currentBox = null;
    const volumeSlider = document.getElementById('volume');

    function playPauseBox() {
      const box = this;
      const fileName = box.dataset.audio;
      if (!fileName) return;

      if (currentBox === box) {
        if (currentAudio.paused) {
          currentAudio.play();
          box.classList.add('playing');
          showPlayingIndicator(box);
        } else {
          currentAudio.pause();
          box.classList.remove('playing');
          removePlayingIndicator(box);
          removeProgressBar(box);
        }
        return;
      }

      if (currentAudio) {
        currentAudio.pause();
        currentAudio.currentTime = 0;
        currentBox.classList.remove('playing');
        removePlayingIndicator(currentBox);
        removeProgressBar(currentBox);
      }

      currentAudio = new Audio(`sounds/${fileName}`);
      currentAudio.loop = false;
      currentAudio.volume = volumeSlider.value;
      currentAudio.play();

      currentBox = box;
      box.classList.add('playing');
      showPlayingIndicator(box);
      showProgressBar(box);

      currentAudio.addEventListener('timeupdate', updateProgressBar);
      currentAudio.addEventListener('ended', () => {
        box.classList.remove('playing');
        removePlayingIndicator(box);
        removeProgressBar(box);
      });
    }

    function showPlayingIndicator(box) {
      let indicator = box.querySelector('.playing-indicator');
      if (!indicator) {
        indicator = document.createElement('div');
        indicator.className = 'playing-indicator';
        indicator.textContent = 'ðŸŽµ Playing...';
        box.appendChild(indicator);
      }
    }

    function removePlayingIndicator(box) {
      const indicator = box.querySelector('.playing-indicator');
      if (indicator) {
        box.removeChild(indicator);
      }
    }

    function showProgressBar(box) {
      let container = box.querySelector('.progress-container');
      if (!container) {
        container = document.createElement('div');
        container.className = 'progress-container';

        const bar = document.createElement('div');
        bar.className = 'progress-bar';
        container.appendChild(bar);

        box.appendChild(container);
      }
    }

    function updateProgressBar() {
      if (!currentAudio || !currentBox) return;
      const bar = currentBox.querySelector('.progress-bar');
      if (bar && currentAudio.duration) {
        const percent = (currentAudio.currentTime / currentAudio.duration) * 100;
        bar.style.width = `${percent}%`;
      }
    }

    function removeProgressBar(box) {
      const container = box.querySelector('.progress-container');
      if (container) {
        box.removeChild(container);
      }
    }

    document.querySelectorAll('.box').forEach(box =>
      box.addEventListener('click', playPauseBox)
    );

    window.addEventListener('keydown', e => {
      if (e.altKey || e.ctrlKey || e.metaKey || e.repeat) return;
      const box = document.querySelector(`.box[data-key="${e.key}"]`);
      if (!box) return;
      e.preventDefault();
      box.click();
    });

    volumeSlider.addEventListener('input', e => {
      if (currentAudio) currentAudio.volume = e.target.value;
    });
  </script>
